\section{Appendix}

    \begin{definition}\label{def:distinguished_strict_affinoid_algebra}
        Let \(A\) be a strictly \(\kk\)-affinoid algebra.
        We say that \(A\) is \emph{distinguished} if there exists a surjective morphism \(\kk\{ x_1, \ldots, x_n \} \to A\) such that the induced norm on \(A\) coincides with the spectral norm \(\rho_A\).
        \Yang{}
    \end{definition}

    \begin{theorem}\label{thm:every_affinoid_algebra_is_distinguished_over_algebraically_closed_field}
        Let \(\kkk\) be an algebraically closed non-Archimedean complete field.
        Then every strictly \(\kkk\)-affinoid algebra is distinguished.
        \Yang{}
    \end{theorem}
