\section{Some conjectures in algebraic dynamics}

Let \(\kk\) be a finitely generated field over \(\bbQ\) and \(\kkk\) be its algebraic closure.

\subsection{Three orbit conjectures}

    \begin{conjecture}\label{conj:Dynamical_Mordell_Lang_conjecture}
        Let \(X\) be a variety over \(\kk\) and \(f: X \ratmap X\) be a dominant rational self-map defined over \(\kk\).
        Let \(x \in X(\kkk)\) be a point such that its orbit \(O_f(x) := \{f^n(x) : n \geq 0\}\) is well-defined and Zariski dense in \(X\).
        Let \(V \subseteq X\) be a subvariety defined over \(\kk\).
        Then the set \(\{n \geq 0 : f^n(x) \in V\}\) is a union of finitely many arithmetic progressions.
        \Yang{}
    \end{conjecture}

    \begin{conjecture}\label{conj:Zariski_dense_orbit_conjecture}
        Let \(X\) be a variety over \(\kk\) and \(f: X \ratmap X\) be a dominant rational self-map defined over \(\kk\).
        If there is no nonconstant rational function \(g \in \kk(X)\) such that \(g \circ f = g\),
        then there exists a point \(x \in X(\kkk)\) whose orbit \(O_f(x) := \{f^n(x) : n \geq 0\}\) is well-defined and Zariski dense in \(X\).
        \Yang{}
    \end{conjecture}

    \begin{conjecture}\label{conj:Kawaguchi_Silverman_conjecture}
        Let \(X\) be a variety over \(\kk\) and \(f: X \ratmap X\) be a dominant rational self-map defined over \(\kk\).
        Let \(x \in X(\kkk)\) be a point whose orbit \(O_f(x) := \{f^n(x) : n \geq 0\}\) is well-defined.
        If the orbit \(O_f(x)\) is Zariski dense in \(X\),
        then the arithmetic degree \(\alpha_f(x)\) exists and equals to the first dynamical degree \(\lambda_1(f)\).
        \Yang{}
    \end{conjecture}
