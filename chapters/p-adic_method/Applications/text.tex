\section{Applications}

\subsection{Existence of non-preperiodic points}

    \begin{theorem}[{ref.\cite[Corollary 9]{Amerik11nonpreperiodic}}]\label{thm:Amerik11_existence_of_nonpreperiodic_points}
        Let \(\kk\) be an algebraically closed field of characteristic \(0\), \(X\) a projective variety defined over \(\kk\), and \(f \colon X \ratmap X\) a dominant rational self-map defined over \(\kk\).
        
        Then there exists a basic adelic subset \(U \subset X(\kk)\) such that the forward orbit \(O_f(x) = \{ f^n(x) : n \geq 0 \}\) is well-defined and infinite for every \(x \in U\).
    \end{theorem}

\subsection{DML conjecture for \'etale morphisms}

    \begin{theorem}[{ref.\cite[Theorem 1.3]{BGT10DMLetale}}]\label{thm:BGT10_DML_for_etale_morphisms}
        Let \(\kk\) be a field of characteristic \(0\), \(X\) a variety defined over \(\kk\), and \(f \colon X \to X\) an \'etale morphism defined over \(\kk\).
        The DML conjecture holds for \((X,f)\).
    \end{theorem}
