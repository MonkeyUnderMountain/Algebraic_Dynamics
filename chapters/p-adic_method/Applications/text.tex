\subsection{Existence of non-preperiodic points}

    \begin{theorem}[{ref.\cite[Corollary 9]{Amerik11nonpreperiodic}}]\label{thm:Amerik11_existence_of_nonpreperiodic_points}
        Let \(\kk\) be an algebraically closed field of characteristic \(0\), \(X\) a projective variety defined over \(\kk\), and \(f \colon X \ratmap X\) a dominant rational self-map defined over \(\kk\).
        Then there exists a point \(x \in X(\kk)\) such that the forward orbit \(O_f(x) = \{ f^n(x) : n \geq 0 \}\) is well-defined and infinite.
    \end{theorem}
